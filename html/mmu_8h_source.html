208 // Set up a normal interrupt/trap gate descriptor.
209 // - istrap: 1 for a trap (= exception) gate, 0 for an interrupt gate.
210 // interrupt gate clears FL_IF, trap gate leaves FL_IF alone
211 // - sel: Code segment selector for interrupt/trap handler
212 // - off: Offset in code segment for interrupt/trap handler
213 // - dpl: Descriptor Privilege Level -
214 // the privilege level required for software to invoke
215 // this interrupt/trap gate explicitly using an int instruction.
216 #define SETGATE(gate, istrap, sel, off, d) \
217 { \
